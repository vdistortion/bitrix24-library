New REST documentation

Install ​

sh
npm i bitrix24-library@latest

Use ​

js
import Bitrix24 from 'bitrix24-library';

Bitrix24.init().then((BX24) => {
  console.log(BX24.getAuth());

  const RestCall = BX24.createBatch();
  RestCall.batch({
    info: ['app.info'],
    profile: ['profile'],
  }).then(console.info);
});
